licenseplate-backend/
├── pom.xml                         // File cấu hình Maven

└── src
    ├── main
    │   ├── java
    │   │   └── com
    │   │       └── example
    │   │           └── licenseplate
    │   │               ├── LicensePlateApplication.java      // Lớp main chạy ứng dụng Spring Boot
    │   │               ├── config
    │   │               │   └── SecurityConfig.java             // Cấu hình bảo mật (JWT, CORS, session…)
    │   │               ├── controller
    │   │               │   ├── AuthController.java             // Endpoint đăng nhập, đăng ký (JWT)
    │   │               │   ├── UserController.java             // CRUD cho người dùng (tất cả)
    │   │               │   ├── AdminController.java            // CRUD cho admin (chỉ admin được phép truy cập)
    │   │               │   └── ScanController.java             // Endpoint nhận dữ liệu quét biển số (upload file, …)
    │   │               ├── dto
    │   │               │   ├── JwtResponse.java                // DTO trả về token và thông tin user
    │   │               │   ├── SignInRequest.java              // DTO đăng nhập
    │   │               │   ├── SignUpRequest.java              // DTO đăng ký cho người dùng
    │   │               │   ├── AdminSignUpRequest.java         // DTO đăng ký cho admin
    │   │               │   └── AdminUpdateRequest.java         // DTO cập nhật thông tin admin
    │   │               ├── filter
    │   │               │   ├── JwtAuthenticationFilter.java    // Bộ lọc xử lý JWT cho request
    │   │               │   └── WhitelistFilter.java            // Bộ lọc cho các URL không cần xác thực
    │   │               ├── model
    │   │               │   ├── ERole.java                      // Enum định nghĩa các vai trò (ROLE_USER, ROLE_ADMIN, ROLE_MODERATOR)
    │   │               │   ├── Role.java                       // Entity ánh xạ bảng roles
    │   │               │   ├── User.java                       // Entity ánh xạ bảng users
    │   │               │   ├── Vehicle.java                    // Entity ánh xạ bảng vehicle (xe hơi, xe 2 bánh)
    │   │               │   └── ScanLog.java                    // Entity ánh xạ bảng scan_log (lịch sử quét)
    │   │               ├── repository
    │   │               │   ├── RoleRepository.java             // Repository cho Role
    │   │               │   ├── UserRepository.java             // Repository cho User
    │   │               │   ├── VehicleRepository.java          // Repository cho Vehicle
    │   │               │   └── ScanLogRepository.java          // Repository cho ScanLog
    │   │               ├── service
    │   │               │   ├── AuthEntryPointJwt.java          // Xử lý lỗi khi xác thực không thành công
    │   │               │   ├── UserDetailsImpl.java            // Cài đặt UserDetails (dùng cho Spring Security)
    │   │               │   ├── UserDetailsServiceImpl.java     // Tải thông tin user từ DB
    │   │               │   ├── UserService.java                // Nghiệp vụ CRUD cho người dùng
    │   │               │   └── ScanService.java                // Nghiệp vụ xử lý quét biển số (lưu file, dữ liệu quét,…)
    │   │               └── util
    │   │                   └── JwtUtil.java                    // Tiện ích tạo và xác thực JWT token
    │   └── resources
    │       └── application.properties                        // Cấu hình ứng dụng (DB, jwt, port,…)
    └── test
        └── java
            └── com
                └── example
                    └── licenseplate
